public static void insertionSort2(int n, List<Integer> arr) {
    // Write your code here
    for(int i = 1; i<n; i++){
        for (int j = i; j>0; j--){
            if(arr.get(j)<arr.get(j-1)){
                Collections.swap(arr, j, j-1);
            }
            
        }
        String p = arr.toString();
        p=p.replace("[", "");
        p=p.replace("]", "");
        p=p.replace(",", "");
        System.out.println(p);
    }
    

    }

}
